登入本機的資料庫
	mysql -u root -p

連接到別人的資料庫
	mysql -h 

查看所有資料庫
	mysql> show databases;

建立資料庫 名稱lccsql 編碼utf8
	mysql> create database lccsql default character set utf8 collate utf8_general_ci; 

使用這個資料庫
	mysql> use lccsql;

建立資料表
	mysql> create table students(
	-> id int primary key auto_increment,
	-> name varchar(50),
	-> sex varchar(2) not null,
	-> age int(2),
	-> birthday date,
	-> address varchar(255));  

查看所有建立的欄位
	mysql> show columns from students; 

選取students內所有資料
	mysql> select * from students; 

在資料表students內建立資料
	mysql> insert into students(name,sex,age,birthday,address) values('Ecco','M',25,'1998-12-08','台中市豐原區萬順一街70號');

面試考題:
	正面:
		1. select max(業績) from 全聯門市業績
		2. select 門市名稱, sum(業績) from 全聯門市業績 group by 門市名稱
		3. select *,(業績/來客數) 客單 from 全聯門市業績 where (業績/來客數)>300
		加入排名欄位
		4. select *, row_number() over(order by 業績 desc) 排名 from 全聯門市業績 order by 業績 desc

	背面:
		無法利用索引的函式: 
			1. 使用 NOT 、!=、not in、not like 等負向判斷詞
			SELECT * FROM users WHERE name != 'John';
			2. 使用函數語句包裝被索引的欄位
			SELECT * FROM users WHERE DATE(created_at) = '2022-02-28';
			3. 資料類型錯誤
			SELECT * FROM users WHERE age = 30;
			4. 使用 『 %XXX% 』或是 『 %XXX 』
			SELECT * FROM users WHERE name like '%john%';
			5. 使用 Null
			SELECT * FROM users WHERE name IS NULL;
			6. 針對索引欄位做運算
			SELECT * FROM users WHERE height - 10 > 150;
		下方題目:
			1. select * from TableA where 銷售量>=500
			2. select * from tablea where 銷售量=(select max(銷售量) from tablea)
			3. select max(銷售量) 最大銷售, min(庫存量) 最小庫存 from TableA
			查詢空值欄位是用is null
			4. select tablea.* from tablea inner join tablec on tablea.門市代號=tablec.門市代號 where tablec.閉店日 is null
			使用substr分隔欄位 拆成1-3和4-6
			5. select substr(tableb.處區 from 1 for 3) as 處, substr(tableb.處區 from 4 for 3) as 區,tableb.門市代號, tableb.門市, tablea.銷售量, tablea.庫存量 from tableb inner join tablea on tablea.門市代號=tableb.門市代號 order by tablea.銷售量 desc

view結合inner join
	mysql> create view mycourse(name, title, area, opendate) as (select student.name, lesson.title, course.area, course.opendate from course inner join student on course.sid = student.id inner join lesson on course.lessonid=lesson.id);
